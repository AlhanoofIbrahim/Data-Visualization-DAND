#get the delays in arrival between the origin and destination city.
plt.figure(figsize = (6,6))
#Top 5 destination cities
top_cities = Flight_clean['DEST_CITY_NAME'].value_counts()[0:5].index
#Create dataframe for top 5 cities 
top_cities = Flight_clean[Flight_clean.DEST_CITY_NAME.isin(top_cities)]
#Heat map
ct_counts = top_cities.groupby(['ORIGIN_CITY_NAME', 'DEST_CITY_NAME']).ARR_DELAY.mean()
ct_counts = ct_counts.reset_index(name='Arrival Delay')
ct_counts = ct_counts.pivot(index = 'DEST_CITY_NAME', columns = 'ORIGIN_CITY_NAME', values = 'Arrival Delay')
sb.heatmap(ct_counts , cmap = "RdYlBu_r")
plt.yticks(rotation = 0)
plt.title('Correlation between top five city destination and origin city based in arrival delay');

We get that the flights between buffalo, NY, and Fayetteville has over 40 flights delays in arrival.
